    @Autowired  
    private DkSaleOrderItemMapper dkSaleOrderItemMapper;  
  
    @Test  
    public void getNoSettleGoodsAmountTest1() {  
        Map<Integer, BigDecimal> noSettleGoodsAmountMap = new HashMap<>();  
  
        NoSettleItemQuery itemQuery = new NoSettleItemQuery();  
        itemQuery.setStallId(10005);  
        itemQuery.setOrderStatusList(CxmOrderStatusQueryConstant.ORDER_STATUS_NOT_EQUAL_REFUND);  
        itemQuery.setExtraAmountFlag(true);  
        itemQuery.setHandFlag(false);  
        List<DkSaleOrderItem> orderItemList = dkSaleOrderItemMapper.getNoSettleGoodsAmountTest1(itemQuery);  
        Map<Integer, List<DkSaleOrderItem>> orderItemMap = orderItemList.stream().collect(Collectors.groupingBy(DkSaleOrderItem::getSupplierId));  
        for( Map.Entry<Integer, List<DkSaleOrderItem>>  orderItemsMap :  orderItemMap.entrySet() ){  
            List<DkSaleOrderItem> orderItems = orderItemsMap.getValue();  
            BigDecimal amount = BigDecimal.ZERO;  
            for( DkSaleOrderItem orderItem : orderItems ){  
                if( orderItem.getOrderStatus() == OrderConstant.ORDER_STATUS_RETURN ){  
                    amount = BigDecimalUtil.subtract(amount, orderItem.getGoodsAmount());  
                } else{  
                    amount = BigDecimalUtil.add(amount, orderItem.getGoodsAmount());  
                }  
            }  
            noSettleGoodsAmountMap.put(orderItemsMap.getKey(), amount);  
        }  
  
    }  
  
    @Test  
    public void getNoSettleGoodsAmountTest2() {  
        Map<Integer, BigDecimal> noSettleGoodsAmountMap = new HashMap<>();  
  
        NoSettleItemQuery itemQuery = new NoSettleItemQuery();  
        itemQuery.setStallId(10005);  
        itemQuery.setOrderStatusList(CxmOrderStatusQueryConstant.ORDER_STATUS_NOT_IN_REFUND_RETURN);  
        itemQuery.setExtraAmountFlag(true);  
        itemQuery.setHandFlag(false);  
        List<DkSaleOrderItem> orderItemList = dkSaleOrderItemMapper.getNoSettleGoodsAmountTest2(itemQuery);  
        Map<Integer, DkSaleOrderItem> orderItemMap = orderItemList.stream().collect(Collectors.toMap( DkSaleOrderItem::getSupplierId, item-> item ));  
        for( Map.Entry<Integer, DkSaleOrderItem> map : orderItemMap.entrySet() ){  
            DkSaleOrderItem orderItem = map.getValue();  
            if(orderItem != null){  
                noSettleGoodsAmountMap.put(map.getKey(), orderItem.getGoodsAmount());  
            }  
        }  
  
        //处理退货  
        itemQuery.setOrderStatusList(null);  
        itemQuery.setOrderStatus(OrderConstant.ORDER_STATUS_RETURN);  
        orderItemList = dkSaleOrderItemMapper.getNoSettleGoodsAmountTest2(itemQuery);  
        orderItemMap = orderItemList.stream().collect(Collectors.toMap( DkSaleOrderItem::getSupplierId, item-> item ));  
        for( Map.Entry<Integer, DkSaleOrderItem> map : orderItemMap.entrySet() ){  
            Integer supplierId = map.getKey();  
            DkSaleOrderItem orderItem = map.getValue();  
            if(orderItem != null){  
                if (noSettleGoodsAmountMap.containsKey(supplierId)) {  
                    //减退货  
                    noSettleGoodsAmountMap.put(supplierId, BigDecimalUtil.subtract(noSettleGoodsAmountMap.get(supplierId), orderItem.getGoodsAmount()));  
                } else {  
                    //退货取反  
                    noSettleGoodsAmountMap.put(supplierId, orderItem.getGoodsAmount().negate());  
                }  
            }  
  
        }  
    }  
}```
