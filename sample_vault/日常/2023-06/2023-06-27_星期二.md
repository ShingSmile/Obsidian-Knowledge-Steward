Map<Integer,Long> settleTimeMap = new HashMap<>();

List<Integer> removeBatchIdList = new ArrayList<>();  
for(Map.Entry<Integer,List<CxmSettlementDoc>> settlementBatch : settlementBatchMap.entrySet()){  
    Integer batchId = settlementBatch.getKey();  
    List<CxmSettlementDoc> list = settlementBatch.getValue();  
    Long settleTime = list.stream().mapToLong(CxmSettlementDoc::getCreateTime).max().orElse(0L);  
    if(settleTime != 0L){  
        settleTimeMap.put(batchId, settleTime);  
    }  
    //筛选结算时间  
    if(query.getTimeSelect() == CxmCommonConstant.CONSTANT_ONE && Objects.nonNull(query.getStartTime()) && Objects.nonNull(query.getEndTime())){  
        if(settleTime < query.getStartTime() || settleTime > query.getEndTime()){  
            removeBatchIdList.add(batchId);  
        }  
    }  
}
